---
title: Configuration
description: Configure the OpenAPI provider, document info, UI endpoints and generator behavior.
---

The configure command creates `config/openapi.ts`. All options live in this file.

```ts
import { defineConfig } from '@outloud/adonis-openapi'
import env from '#start/env'
import packageJson from '../package.json' with { type: 'json' }

const openapiConfig = defineConfig({
  enabled: true,
  provider: 'scalar',
  document: {
    info: {
      title: env.get('APP_DISPLAY_NAME', packageJson.name),
      version: packageJson.version,
    },
    servers: [
      { url: env.get('APP_URL', 'http://localhost:' + env.get('PORT')) },
    ],
  },
  generator: {
    resolve: import.meta.resolve,
    routes: true,
  },
})

export default openapiConfig
```

## Options

::field-group
  ::field{name="enabled" type="boolean"}
    Master toggle. When `false`, the provider does not register the `/docs` or `/openapi.json` routes.

    ```ts
    enabled: env.get('NODE_ENV') !== 'production'
    ```
  ::

  ::field{name="provider" type="'scalar'"}
    UI provider used to render the API reference. Currently `scalar` is the only built-in option.
  ::

  ::field{name="scalar" type="ScalarOptions | undefined"}
    Options forwarded to the Scalar viewer.

    ```ts
    scalar: {
      proxy: true, // use default Scalar proxy
      // proxy: 'https://my-proxy.example' // or a custom one
    }
    ```
  ::

  ::field{name="endpoints.ui" type="string | undefined"}
    URL path where the UI is served.
    <br>
    <br>Defaults to `/docs`.

    ```ts
    endpoints: { ui: '/api-reference' }
    ```
  ::

  ::field{name="endpoints.spec" type="string | undefined"}
    URL path where the raw `openapi.json` is served.
    <br>
    <br>Defaults to `/openapi.json`.

    ```ts
    endpoints: { spec: '/api/openapi.json' }
    ```
  ::

  ::field{name="document" type="OpenAPIV3_1.Document"}
    Extend OpenAPI document, it will be merged with generated schemas and paths.

    Use it for `info`, `servers`, `tags`, `security`, `components.securitySchemes`, etc.

    ```ts
    document: {
      info: { title: 'My API', version: '1.0.0' },
      servers: [{ url: 'https://api.example.com' }],
      components: {
        securitySchemes: {
          bearer: { type: 'http', scheme: 'bearer' },
        },
      },
      security: [{ bearer: [] }],
    }
    ```
  ::

  ::field{name="generator.resolve" type="(specifier, parent?) => string"}
    Module resolver used to map type-import specifiers to file paths. Always set to `import.meta.resolve`.

    ```ts
    generator: { resolve: import.meta.resolve }
    ```
  ::

  ::field{name="generator.routes" type="true | 'auto' | { include?, exclude? }"}
    Controls which routes are included in the spec.

    - `true` — include every route (default).
    - `'auto'` — include only routes that have a `summary` declared via `.openapi({ summary: ... })`.
    - `{ include: [...] }` — only the listed route patterns (e.g. `/users/:id`).
    - `{ exclude: [...] }` — every route except the listed patterns.

    ```ts
    generator: {
      routes: { exclude: ['/health', '/metrics'] },
    }
    ```
  ::

  ::field{name="generator.handlers" type="Handler[] | undefined"}
    Extra type → schema handlers, prepended to the built-in pipeline. Use them to teach the generator about your own types or to override defaults (e.g. emit `DateTime` as `date-time`). See [Custom Handlers](/guide/handlers).

    ```ts
    import type { Handler } from '@outloud/adonis-openapi-generator'

    const decimalHandler: Handler = {
      path: /\/decimal\.js/,
      name: 'Decimal',
      schema: () => ({ type: 'string' }),
    }

    generator: {
      resolve: import.meta.resolve,
      handlers: [decimalHandler],
    }
    ```
  ::
::

## Disabling in Specific Environments

```ts
defineConfig({
  enabled: env.get('NODE_ENV') !== 'production',
  // ...
})
```

::warning
When `enabled` is `false`, calls to `.openapi()` on routes still type-check and execute but no routes are registered to serve the spec or UI.
::

## Merging With Generated Spec

Anything you place under `document` is merged on top of the generated paths and components:

```ts
defineConfig({
  document: {
    info: { title: 'Users API', version: '2.0.0' },
    paths: {
      '/health': {
        get: { summary: 'Health check', responses: { 200: { description: 'OK' } } },
      },
    },
  },
  // ...
})
```

The `/health` operation will appear in the final spec even if it is filtered out of the generated routes.
